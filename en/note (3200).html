FTP Client : Fedora
2015/06/03
 
For how to connect to FTP server from Clients computer, the example follows is on Fedora.
[1] Install FTP Client.
[root@dlp ~]#
dnf -y install lftp
[2] The connection with root account is prohibited by default, so access with an common user to FTP Server.
# lftp [option] [hostname]

[redhat@dlp ~]$
lftp -u fedora www.srv.world

Password:
# password of the user

lftp fedora@www.srv.world:~>
# show current directory on FTP server

lftp fedora@www.srv.world:~>
pwd

ftp://fedora@www.srv.world
# show current directory on local server

lftp fedora@www.srv.world:~>
!pwd

/home/redhat
# show files in current directory on FTP server

lftp fedora@www.srv.world:~>
ls

drwxr-xr-x    2 1000     1000           23 Dec 19 01:33 public_html
-rw-r--r--    1 1000     1000          399 Dec 20 16:32 test.py
# show files in current directory on local server

lftp fedora@www.srv.world:~>
!ls -l

total 12
-rw-rw-r-- 1 redhat redhat 10 Dec 20 14:30 redhat.txt
-rw-rw-r-- 1 redhat redhat 10 Dec 20 14:59 test2.txt
-rw-rw-r-- 1 redhat redhat 10 Dec 20 14:59 test.txt
# change directory

lftp fedora@www.srv.world:~>
cd public_html

lftp fedora@www.srv.world:~/public_html>
pwd

ftp://fedora@www.srv.world/%2Fhome/fedora/public_html
# upload a file to FTP server

# "-a" means ascii mode ( default is binary mode )

lftp fedora@www.srv.world:~>
put -a redhat.txt test.txt

22 bytes transferred
Total 2 files transferred
lftp fedora@www.srv.world:~>
ls

drwxr-xr-x    2 1000     1000           23 Dec 19 01:33 public_html
-rw-r--r--    1 1000     1000           10 Dec 20 17:01 redhat.txt
-rw-r--r--    1 1000     1000          399 Dec 20 16:32 test.py
-rw-r--r--    1 1000     1000           10 Dec 20 17:01 test.txt
# upload some files to FTP server

lftp fedora@www.srv.world:~>
mput -a test.txt test2.txt

22 bytes transferred
Total 2 files transferred
lftp fedora@www.srv.world:~>
ls

drwxr-xr-x    2 1000     1000           23 Dec 19 01:33 public_html
-rw-r--r--    1 1000     1000          399 Dec 20 16:32 test.py
-rw-r--r--    1 1000     1000           10 Dec 20 17:06 test.txt
-rw-r--r--    1 1000     1000           10 Dec 20 17:06 test2.txt
# download a file from FTP server

# "-a" means ascii mode ( default is binary mode )

lftp fedora@www.srv.world:~>
get -a test.py

416 bytes transferred
# download some files from FTP server

lftp fedora@www.srv.world:~>
mget -a test.txt test2.txt

20 bytes transferred
Total 2 files transferred

lftp fedora@www.srv.world:~>
mkdir testdir

lftp fedora@www.srv.world:~>
ls
