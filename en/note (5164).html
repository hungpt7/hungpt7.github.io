Install KVM + QEMU
  It's the Virtualization settings with KVM ( Kernel-based Virtual Machine ) + QEMU. This requires that the CPU on your computer has a function Intel VT or AMD-V.

[1] Install KVM.
[root@dlp ~]#
yum -y install qemu-kvm qemu bridge-utils


[root@dlp ~]#
modprobe kvm

[root@dlp ~]#
modprobe kvm_intel
 
# if AMD, "kvm_amd"

[root@dlp ~]#
lsmod | grep kvm

kvm_intel
47768 0

kvm
152904 1 kvm_intel
[2] Configure Bridge networking.
[root@dlp ~]#
cd /etc/sysconfig/network-scripts

[root@dlp network-scripts]#
cp ifcfg-eth0 ifcfg-br0

[root@dlp network-scripts]#
vi ifcfg-br0


DEVICE=
br0
# change

HWADDR=00:24:1D:13:7A:26
ONBOOT=yes
IPADDR=192.168.0.20
BOOTPROTO=none
NETMASK=255.255.255.0
TYPE=
Bridge
# change

GATEWAY=192.168.0.1
DNS1=192.168.0.10
IPV6INIT=no
USERCTL=no
PREFIX=24
NAME="System eth0"
UUID=5fb06bd0-0bb0-7ffb-45f1-d6edd65f3e03

[root@ns network-scripts]#
vi ifcfg-eth0


DEVICE=eth0
HWADDR=00:24:1D:13:7A:26
ONBOOT=yes
IPADDR=192.168.0.20
BOOTPROTO=none
NETMASK=255.255.255.0
TYPE=Ethernet
GATEWAY=192.168.0.1
DNS1=192.168.0.10
IPV6INIT=no
USERCTL=no
PREFIX=24
NAME="System eth0"
UUID=5fb06bd0-0bb0-7ffb-45f1-d6edd65f3e03
BRIDGE=br0
# add


[root@dlp network-scripts]#
cd

[root@dlp ~]#
/etc/rc.d/init.d/network restart
