Basic Operation
2016/12/12
 
Basic Operarions with virsh command.
[1] Start Virtual Machine
# Start Virtual Machine 'fedora25'

[root@dlp ~]#
virsh start fedora25

Domain fedora25 started
# start and connect to console of 'fedora25'

[root@dlp ~]#
virsh start fedora25 --console

Domain fedora25 started
Connected to domain fedora25
[2] Stop Virtual Machine
# Stop Virtual Machine 'fedora25'

[root@dlp ~]#
virsh shutdown fedora25

Domain fedora25 is being shutdown
# Stop Virtual Machine 'fedora25' forcely

[root@dlp ~]#
virsh destroy fedora25

Domain fedora25 destroyed
[3] Set auto-start for Virtual Machines
# Enable auto-start for 'fedora25'

[root@dlp ~]#
virsh autostart fedora25

Domain fedora25 marked as autostarted
# Disable auto-start for 'fedora25'

[root@dlp ~]#
virsh autostart --disable fedora25

Domain fedora25 unmarked as autostarted
[4] List all Virtual Machines
# List all active Virtual Machines

[root@dlp ~]#
virsh list 

 Id    Name               State
----------------------------------------
 2     fedora25            running
# List all Virtual Machines included inactives

[root@dlp ~]#
virsh list --all 

 Id    Name               State
----------------------------------------
 -     fedora25            shut off
 -     template           shut off
 -     Win2k12            shut off
[5] Switch console,
Move to GuestOS to HostOS with Ctrl + ] key.
Move to HostOS to GuestOS with a command 'virsh console (name of virtual machine)'.
# connect to 'fedora25'

[root@dlp ~]#
virsh console fedora25

Connected to domain fedora25
Escape character is ^]    
# Enter key

localhost login:

[root@localhost ~]#    
# Ctrl + ] key

[root@dlp ~]#    
# just switched on Host
