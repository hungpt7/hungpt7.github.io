Install MySQL
2012/06/25
  Install MySQL to configure Database Server.
root@www:~#
aptitude -y install mysql-server-5.5
# set MySQL's root password

root@www:~#
mysql -u root -p
# connect to MySQL

Enter password:
# MySQL root password
