Samba BDC
2011/03/12
  Configure Samba BDC (backup domain controller). Of course, Samba PDC is required to be running in your LAN and also this Samba BDC is LDAP client.
[1] smb.conf is almost the same with PDC. Different section is 'domain master' section.
root@slave:~#
aptitude -y install smbldap-tools


root@slave:~#
mv /etc/samba/smb.conf /etc/samba/smb.conf.bak

root@slave:~#
cp /usr/share/doc/smbldap-tools/examples/smb.conf /etc/samba/smb.conf

root@slave:~#
vi /etc/samba/smb.conf


# line 3: change workgroup name to any one

workgroup =
ServerWorld


# line 12: maike it comment

#
min passwd length = 3

# line 22: change

ldap passwd sync =
yes


# line 33,34: change

Dos charset =
CP932

Unix charset =
UTF-8


# line 42: change (different section from PDC)

domain master =
No


# line 47: change (specify LDAP server)

passdb backend = ldapsam:
ldap://10.0.0.100/


# line 48: change LDAP admin DN (LDAP server's one)

ldap admin dn =
cn=admin,dc=server,dc=world


# line 50: change LDAP suffix (LDAP server's one)

ldap suffix =
dc=server,dc=world

ldap group suffix = ou=
groups

ldap user suffix = ou=
people


# line 60: uncomment

delete group script = /usr/sbin/smbldap-groupdel "%g"

# line 64: add (specify admin user, so SSL)

set primary group script = /usr/sbin/smbldap-usermod -g '%g' '%u'
admin users = domainadm
ldap ssl = no


root@slave:~#
mkdir /home/netlogon

root@slave:~#
/etc/init.d/samba restart

Stopping Samba daemons: nmbd smbd.
Starting Samba daemons: nmbd smbd.

root@slave:~#
smbpasswd -W
# add LDAP admin's password

Setting stored password for "cn=admin,dc=server,dc=world" in secrets.tdb
New SMB password:
# LDAP admin password

Retype new SMB password:

root@slave:~#
net rpc getsid
# get SID in PDC

Storing SID S-1-5-21-2752024775-1437179205-4226352253 for Domain SERVERWORLD in secrets.tdb  
# remember this


root@slave:~#
gzip -d /usr/share/doc/smbldap-tools/configure.pl.gz

root@slave:~#
perl /usr/share/doc/smbldap-tools/configure.pl
