Configure normal resolving
  Create zone files that servers resolve IP address from domain name.
 
[1] For internal zone
This example uses internal address[10.0.0.0/24], domain name[srv.world], but please use your own one when you set config on your server.
[root@dlp ~]#
vi /var/named/srv.world.lan


$TTL
86400

@
IN
SOA
dlp.srv.world. root.srv.world. (

2009111801
;Serial

3600
;Refresh

1800
;Retry

604800
;Expire

86400
;Minimum TTL

)
# define name server

IN
NS
dlp.srv.world.

# define name server's IP address

IN
A
10.0.0.30

# define mail exchanger

IN
MX 10
dlp.srv.world.

10.0.0.30
