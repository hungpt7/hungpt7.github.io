Mail Log Analyzer - pflogsumm
2011/05/04
  Install pflogsumm that is Postfix mail log analyzer.
root@mail:~#
aptitude -y install pflogsumm


# generate log summary for yesterday

root@mail:~#
perl /usr/sbin/pflogsumm -d yesterday /var/log/mail.log
