Basic Operation
2017/06/22
 
This is the basic Operarions with virsh command.
[1] Start Virtual Machine.
# start Virtual Machine [debian]

root@dlp:~#
virsh start debian

Domain debian started
# start and connect to console of [debian]

root@dlp:~#
virsh start debian --console

Domain debian started
Connected to domain debian
[2] Stop Virtual Machine.
# stop Virtual Machine [debian]

root@dlp:~#
virsh shutdown debian

Domain debian is being shutdown
# stop fourcely Virtual Machine [debian]

root@dlp:~#
virsh destroy debian

Domain debian destroyed
[3] Set auto-start for Virtual Machines.
# enable auto-start for [debian]

root@dlp:~#
virsh autostart debian

Domain debian marked as autostarted
# disable auto-start for [debian]

root@dlp:~#
virsh autostart --disable debian

Domain debian unmarked as autostarted
[4] List all Virtual Machines.
# list all active Virtual Machines

root@dlp:~#
virsh list

  Id Name     State
------------------------------------------
  2   win2k12R2     running
# list all Virtual Machines include inactives

root@dlp:~#
virsh list --all

  Id Name     State
------------------------------------------
  2   win2k12R2     running
  1   debian     shut off
[5] Switch console,
Move to GuestOS to HostOS with Ctrl + ] key.
Move to HostOS to GuestOS with a command [virsh console (name of virtual machine)].
root@dlp:~#
virsh console debian
   
# connect to [debian]

Connected to domain debian
Escape character is ^]    
# Enter key
Debian GNU/Linux 9 debian ttyS0
debian login:    
# just switched on Guest
Password:
root@debian:~#    
# Ctrl + ] key

root@dlp:~#    
# just switched on Parent Host
[6] For Other options, There are many options, please try to execute them.
root@dlp:~#
